It is given that
$$
\begin{cases}
\log_3 x + \log_3 y = 9 \\
\log_x 81 - \log_y 9 = 1
\end{cases}
, \quad \text{where } 0<x<y.
$$
\begin{enumx}[label=(\alph*)]
    \item Let $u=\log_3 y$. Prove that $u^2-3u-18=0$.
	\marks{2}
    \item Find $x$.
	\marks{2}
\end{enumerate}
